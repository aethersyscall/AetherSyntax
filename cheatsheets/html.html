HTML 5

1. Document Structure

<!DOCTYPE html> <!-- Defines HTML5 document -->
<html lang="en"> <!-- Root element -->
<head>
    <meta charset="UTF-8"> <!-- Character encoding -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- Responsive -->
    <title>Page Title</title>
    <link rel="icon" href="favicon.ico">
</head>
<body>
    <!-- Content goes here -->
</body>
</html>

2. Text Formatting

<!-- Headings (h1 to h6) -->
<h1>Main Heading</h1>
<h2>Subheading</h2>

<!-- Paragraphs & Breaks -->
<p>This is a paragraph.</p>
<br> <!-- Line break (self-closing) -->
<hr> <!-- Horizontal Rule (line) -->

<!-- Inline Formatting -->
<strong>Bold (Important)</strong>
<b>Bold (Visual)</b>
<em>Italic (Emphasis)</em>
<i>Italic (Visual)</i>
<u>Underline</u>
<s>Strikethrough</s>
<mark>Highlighted</mark>
<code>Code snippet</code>
<sub>Subscript</sub> and <sup>Superscript</sup>

<!-- Generic Containers -->
<div>Block-level container</div>
<span>Inline container</span>

3. Inline CSS

Use the style attribute to apply CSS directly to an element. Syntax: style="property: value; property: value;"

<!-- Text Styling -->
<h1 style="color: blue; text-align: center;">Centered Blue Title</h1>
<p style="font-size: 18px; font-weight: bold; font-family: Arial;">Styled Text</p>

<!-- Backgrounds & Borders -->
<div style="background-color: #f0f0f0; border: 1px solid black; padding: 10px;">
    Box content
</div>

<!-- Dimensions & Spacing -->
<img src="logo.png" style="width: 100px; height: auto; margin-bottom: 20px;">

<!-- Visibility -->
<div style="display: none;">Hidden</div>
<div style="display: block;">Visible</div>

<!-- Layout (Flexbox inline) -->
<div style="display: flex; justify-content: space-between;">
    <div>Left</div>
    <div>Right</div>
</div>

4. Links & Media

<!-- Anchors (Links) -->
<a href="https://google.com">Visit Google</a>
<a href="about.html" target="_blank">Open in new tab</a>
<a href="mailto:me@example.com">Send Email</a>
<a href="#section1">Jump to ID</a>

<!-- Images -->
<img src="image.jpg" alt="Description" width="200" height="200">

<!-- Audio -->
<audio controls>
    <source src="song.mp3" type="audio/mpeg">
</audio>

<!-- Video -->
<video width="320" height="240" controls poster="thumb.jpg">
    <source src="movie.mp4" type="video/mp4">
</video>

5. Inline JavaScript (Events)

Execute JavaScript directly within HTML attributes. Syntax: event="javascript_code"

<!-- Click Events -->
<button onclick="alert('Hello World!')">Click Me</button>
<button onclick="this.innerHTML = 'Clicked!';">Change Text</button>

<!-- Input Events -->
<input type="text" oninput="console.log(this.value)" placeholder="Type here...">
<input type="text" onchange="alert('Value changed!')">

<!-- Mouse Events -->
<div onmouseover="this.style.backgroundColor='red'" 
     onmouseout="this.style.backgroundColor='white'">
    Hover over me
</div>

<!-- Form Submission -->
<form onsubmit="alert('Submitting!'); return false;"> <!-- return false prevents reload -->
    <button type="submit">Submit</button>
</form>

<!-- Window Events (on Body) -->
<body onload="console.log('Page Loaded')">

6. Lists

<!-- Unordered List (Bullets) -->
<ul>
    <li>Item 1</li>
    <li>Item 2</li>
</ul>

<!-- Ordered List (Numbers) -->
<ol type="1"> <!-- types: 1, A, a, I, i -->
    <li>First</li>
    <li>Second</li>
</ol>

<!-- Definition List -->
<dl>
    <dt>HTML</dt>
    <dd>HyperText Markup Language</dd>
</dl>

7. Tables

<table border="1" style="border-collapse: collapse; width: 100%;">
    <thead>
        <tr>
            <th>Name</th>
            <th>Age</th>
        </tr>
    </thead>
    <tbody>
        <tr>
            <td>Alice</td>
            <td>25</td>
        </tr>
        <tr>
            <!-- Colspan merges columns -->
            <td colspan="2">No more data</td>
        </tr>
    </tbody>
</table>

8. Forms & Inputs

<form action="/submit" method="POST">
    <!-- Label -->
    <label for="fname">First name:</label>
    
    <!-- Text Input -->
    <input type="text" id="fname" name="fname" placeholder="John" required>
    
    <!-- Password -->
    <input type="password" name="pass" minlength="8">
    
    <!-- Number -->
    <input type="number" name="age" min="1" max="100" value="18">
    
    <!-- Radio Buttons (Single choice) -->
    <input type="radio" name="gender" value="male"> Male
    <input type="radio" name="gender" value="female"> Female
    
    <!-- Checkbox (Multiple choice) -->
    <input type="checkbox" name="newsletter" checked> Subscribe
    
    <!-- Dropdown -->
    <select name="cars">
        <option value="volvo">Volvo</option>
        <option value="saab">Saab</option>
    </select>
    
    <!-- Multi-line text -->
    <textarea rows="4" cols="50">Enter message...</textarea>
    
    <!-- Buttons -->
    <input type="submit" value="Submit">
    <button type="reset">Reset</button>
</form>

9. Semantic HTML5 (Structural)

Better for SEO and Screen Readers than using <div> for everything.

<header> <!-- Logo, Nav --> </header>

<nav> <!-- Navigation links --> </nav>

<main> <!-- Dominant content -->
    <section> <!-- Thematic grouping -->
        <article> <!-- Self-contained content (blog post) --> </article>
    </section>
    
    <aside> <!-- Sidebar, ads --> </aside>
</main>

<footer> <!-- Copyright, footer links --> </footer>
