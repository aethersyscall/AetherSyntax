C++ Cheatsheet

1. Basic Structure

#include <iostream> // Input/Output streams
#include <vector>   // STL Container

// Standard namespace (avoids typing std:: repeatedly)
using namespace std; 

int main() {
    cout << "Hello, World!" << endl;
    return 0;
}

2. Variables & Initialization

// Explicit types
int a = 5;
double b = 3.14;
bool c = true;
char d = 'A';

// Type Inference (auto)
auto x = 10;        // int
auto y = 5.5;       // double
auto& z = x;        // int& (reference)

// Uniform Initialization (C++11) - Prevents narrowing conversions
int n {5};          // Preferred in modern C++
string s {"Hello"};
int empty {};       // Initializes to 0

// References (Aliases)
int original = 10;
int& ref = original; // ref references original
ref = 20;            // original is now 20

3. Input & Output (<iostream>)

int age;
cout << "Enter age: ";      // Print to console
cin >> age;                 // Read from console

// Error stream
cerr << "Error occurred!";

// Format manipulators (<iomanip>)
cout << fixed << setprecision(2) << 3.14159; // 3.14

4. Control Flow

// If-Else with Initializer (C++17)
if (int val = getValue(); val > 0) {
    cout << val;
} else {
    cout << "Invalid";
}

// Range-based For Loop (Foreach)
vector<int> nums = {1, 2, 3};
for (int n : nums) { cout << n; }       // Copy
for (const auto& n : nums) { cout << n; } // Read-only Reference (Efficient)

// Switch
switch (x) {
    case 1: break;
    case 2: [[fallthrough]]; // C++17 attribute
    default: break;
}

5. Functions

// Default Arguments
int add(int a, int b = 10) { return a + b; }

// Pass by Reference (Modifies original)
void increment(int& n) { n++; }

// Pass by Const Reference (Read-only, avoids copying large objects)
void printString(const string& str) { cout << str; }

// Function Overloading
void print(int i) { cout << "Int: " << i; }
void print(double d) { cout << "Double: " << d; }

// Inline (Suggestion to compiler to embed code)
inline int square(int x) { return x * x; }

6. Strings (<string>)

string s1 = "Hello";
string s2 = "World";

// Concatenation
string s3 = s1 + " " + s2;

// Access
char c = s1[0]; // 'H'
s1.at(0);       // 'H' (Bounds checked)

// Modification
s1.push_back('!');
s1.append(" There");
s1.length();    // or s1.size()

// C++17 String View (Lightweight, non-owning reference)
#include <string_view>
void analyze(string_view sv) { /* No copy made */ }

7. Classes & OOP

class Rectangle {
private:
    int width, height; // Private members

public:
    // Constructor (Initializer List is preferred)
    Rectangle(int w, int h) : width(w), height(h) {}

    // Method
    int area() const { // 'const' means it won't modify members
        return width * height;
    }
    
    // Destructor
    ~Rectangle() { /* Cleanup */ }
};

// Inheritance
class Square : public Rectangle {
public:
    Square(int size) : Rectangle(size, size) {}
};

8. Polymorphism

class Animal {
public:
    // Virtual function allows override
    virtual void speak() const { cout << "???"; }
    
    // Virtual Destructor (Crucial for base classes)
    virtual ~Animal() = default; 
};

class Dog : public Animal {
public:
    // 'override' keyword ensures you are actually overriding
    void speak() const override { cout << "Woof"; }
};

// Usage
Animal* a = new Dog();
a->speak(); // Prints "Woof"
delete a;

9. Memory Management (Smart Pointers)

Forget new and delete (mostly). Use <memory>.

// unique_ptr: Exclusive ownership (Automatically deleted when out of scope)
std::unique_ptr<int> p1 = std::make_unique<int>(10);
// auto p2 = p1; // ERROR: Cannot copy unique_ptr
auto p2 = std::move(p1); // OK: Ownership transferred to p2, p1 is now null

// shared_ptr: Shared ownership (Reference counted)
std::shared_ptr<int> sp1 = std::make_shared<int>(20);
auto sp2 = sp1; // OK: Ref count = 2. Deleted only when all owners are gone.

// weak_ptr: Non-owning reference (Breaks circular dependencies)
std::weak_ptr<int> wp = sp1;

10. STL Containers

// Vector (Dynamic Array)
vector<int> v = {1, 2, 3};
v.push_back(4);
v[0]; // Access

// Array (Fixed size array, safer than C-arrays)
array<int, 3> arr = {1, 2, 3};

// Map (Key-Value pairs, sorted by key)
map<string, int> ages;
ages["Alice"] = 30;
if (ages.count("Alice")) { /* Exists */ }

// Unordered Map (Hash Map - O(1) access)
unordered_map<string, int> scores;

// Pair & Tuple
pair<int, string> p = {1, "Apple"};
tuple<int, double, int> t = {1, 3.14, 99};

11. Iterators & Algorithms (<algorithm>)

vector<int> v = {5, 2, 8, 1};

// Sorting
sort(v.begin(), v.end()); // 1, 2, 5, 8

// Find
auto it = find(v.begin(), v.end(), 8);
if (it != v.end()) { /* Found */ }

// Transform (Map)
transform(v.begin(), v.end(), v.begin(), [](int n){ return n * 2; });

12. Lambdas (Anonymous Functions)

// [capture](parameters) -> return_type { body }

int x = 10;
auto addX = [x](int a) { return a + x; }; // Capture x by value
auto modX = [&x]() { x = 20; };           // Capture x by reference

// Generic Lambda (C++14)
auto print = [](auto val) { cout << val; };

13. Templates (Generics)

// Function Template
template <typename T>
T add(T a, T b) {
    return a + b;
}
int sum = add<int>(5, 10);

// Class Template
template <typename T>
class Box {
    T data;
public:
    Box(T d) : data(d) {}
};
Box<int> intBox(123);

14. Modern C++ Features

// nullptr (Type-safe null pointer)
int* p = nullptr; 

// Structured Binding (C++17) - Unpack tuples/structs
pair<int, int> coords = {10, 20};
auto [x, y] = coords; 

// Optional (C++17) - Value may or may not exist
std::optional<int> getValue(bool valid) {
    if (valid) return 5;
    return std::nullopt;
}

// Variant (C++17) - Type-safe union
std::variant<int, string> data;
data = 10;
data = "Hello";

// Constexpr (Compile-time evaluation)
constexpr int fib(int n) { return (n <= 1) ? n : fib(n-1) + fib(n-2); }
int val = fib(10); // Calculated at compile time

15. Move Semantics (Advanced)

Crucial for performance (avoids deep copies).

// L-value: Has a name, address (int a)
// R-value: Temporary, no address (5, x + y)

// Move Constructor
class Buffer {
    int* data;
public:
    // Move Ctor: Steals resources from 'other'
    Buffer(Buffer&& other) noexcept : data(other.data) {
        other.data = nullptr; // Nullify source so destructor doesn't free it
    }
};

string a = "Heavy String";
string b = std::move(a); // 'a' is now empty/valid state, 'b' owns the data
