1. Basic Structure

#include <stdio.h> // Standard Input/Output
            library

2. Variables & Data Types

// Primitive Types
        int a = 10; //
          Integer (usually 4 bytes)
        float b = 3.14f; //
          Float (4 bytes)
        double c = 3.14159; // Double precision (8 bytes)
        char d = 'A'; //
          Character (1 byte)

        // Modifiers
        unsigned int u = 10; // Positive only
        long long l = 1000000; // Extended range (at least 8 bytes)
        short s = 5; //
          Smaller range (usually 2 bytes)
        const int CONSTANT = 100; // Read-only

        // Booleans (requires C99+)
        #include <stdbool.h>
        bool isValid = true; // true (1) or false (0)

3. Input & Output (printf/scanf)

// Format Specifiers:
        // %d (int), %f (float), %lf (double), %c (char), %s (string), %p (pointer), %x
          (hex)

        int age;
        float height;

        printf("Age: %d, Height: %.2f\n", 25, 1.75); // .2f
          limits decimal places

        // Input (Note the & address-of operator)
        scanf("%d", &age);
        scanf("%f", &height);

5. Arrays & Strings

        int numbers[5] = {1, 2, 3, 4, 5};

        numbers[0] = 10;

        char str1[] = "Hello"; // Size is 6 (5 chars + '\0')

6. Pointers (The Core of C)

int x = 10;

        // Declaration & Initialization
        int *ptr = &x; // ptr stores the address of
          x
        // & is "Address of" operator

        // Dereferencing
        int val = *ptr; // Get value at address (val =
          10)
        *ptr = 20; // Change value at address (x becomes
          20)

        // Null Pointer
        int *p = NULL; //
          Point to nothing (safest default)

        // Generic Pointer
        void *gp = &x; // Can hold any address, needs cast
          to dereference

7. Pointer Arithmetic & Arrays

int arr[3] =
        {10, 20, 30};
        int *p = arr; // Array name decays to pointer to first
          element

        int first = *p; // 10
        int second = *(p + 1); // 20 (moves pointer sizeof(int) bytes forward)

        // Array access using pointers
        p[1] == *(p + 1); // True

8. Dynamic Memory (Heap)

Requires <stdlib.h>. You must manually manage memory.

// malloc: Allocate bytes (uninitialized)
        int *arr = (int*)malloc(5 * sizeof(int));

        if (arr == NULL) { /* Allocation failed */ }

        // calloc: Allocate and zero-initialize
        int *zero_arr = (int*)calloc(5, sizeof(int));

        // realloc: Resize existing memory block
        arr = (int*)realloc(arr, 10 * sizeof(int));

        // free: Release memory (Crucial to prevent leaks)
        free(arr);
        arr = NULL; // Good practice to prevent dangling
          pointer

9. Functions

// Declaration (Prototype)
        int add(int x, int y);

        // Definition
        int add(int x, int y) {
        return x + y;
        }

        // Pass by Value (Default)
        void modifyVal(int a) { a = 99; }
        // Doesn't affect caller

        // Pass by Reference (using Pointers)
        void modifyRef(int *a) { *a = 99;
        } // Affects caller
        // Call: modifyRef(&val);

10. Structs & Unions

// Struct definition
        struct Point {
        int x;
        int y;
        };

        // Initialization
        struct Point p1 = {10, 20};

        // Accessing fields
        p1.x = 15;

        // Pointers to Structs
        struct Point *ptr = &p1;
        ptr->x = 30; // Arrow operator (equivalent to
          (*ptr).x)

        // Typedef (Aliases)
        typedef struct
          {
        char name[50];
        int age;
        } Person; // "Person" is now the type name

        Person person1;

        // Union (Shared memory location for all members)
        union Data {
        int i;
        float f;
        char str[20];
        }; // Size is size of largest member

13. File I/O

FILE *fptr;

        // Open file
        fptr = fopen("file.txt", "w"); // "w" (write), "r" (read), "a" (append)

        if (fptr == NULL) { /* Error */ }

        // Write
        fprintf(fptr, "Value: %d", 100);
        fputs("Hello World", fptr);

        // Read
        char buffer[100];
        fgets(buffer, 100, fptr); // Reads line
        fscanf(fptr, "%s", buffer);

        // Close
        fclose(fptr);

14. Storage Classes

// Global (available everywhere)
        int globalVar;

        // Static (Global lifetime, local scope)
        void counter() {
        static int count = 0; // Initialized only once
        count++;
        }

        // Extern (Defined in another file)
        extern int sharedVar;

        // Volatile (Prevent compiler optimization - mostly embedded)
        volatile int hardware_register;

15. Bitwise Operators

          0011

        int and = a & b; // 0001 (1)
        int or = a | b; // 0111 (7)
        int xor = a ^ b; // 0110 (6)
        int not = ~a; // 1010... (inverted bits)
