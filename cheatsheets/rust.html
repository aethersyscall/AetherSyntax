Rust Cheatsheet

1. Variables & Types

// Immutable variable (default)
        let x = 5;

        // Mutable variable
        let mut y
        = 5;
        y = 10;

        // Constants (MUST have type, uppercase convention)
        const MAX_POINTS: u32 = 100_000;

        // Static variables (global, fixed address)
        static HELLO: &str = "Hello, world!";

        // Type annotation
        let guess: u32 = "42".parse().expect("Not a number!");

        // Shadowing
        let x = 5;
        let x = x + 1; // x is now 6, new variable

2. Primitive Types

// Scalar Types
        let i: i32 =
        -42; // Signed: i8, i16, i32, i64, i128,
          isize
        let u: u32 =
        42; // Unsigned: u8, u16, u32, u64, u128,
          usize
        let f: f64 =
        3.14; // Floats: f32, f64
        let b: bool
        = true; // Booleans: true, false
        let c: char
        = 'z'; // Character (4 bytes, Unicode)

        // Compound Types
        let tup: (i32, f64, u8) = (500, 6.4, 1);
        // Tuple
        let (x, y, z) = tup; // Destructuring
        let val = tup.0; // Access by index

        let arr: [i32; 5] = [1, 2, 3, 4, 5]; // Array (fixed size, stack)
        let same: [i32; 5] = [3; 5]; // [3, 3, 3, 3, 3]
        let first = arr[0]; // Access

4. Control Flow

// If expression
        let number = 3;
        if number < 5 {
        println!("True");
        } else if number == 5 {
        println!("Equal");
        } else {
        println!("False");
        }

        // If as an expression
        let condition = true;
        let number = if condition { 5 } else { 6 };

        // Loops
        loop { println!("Infinite"); break; } // "break" to exit

        let mut count = 0;
        'label: loop { //
          Loop labels
        count += 1;
        if count == 2 { continue 'label; } //
          Skip iteration
        if count == 5 { break 'label; } //
          Break outer loop
        }

        while number != 0 { /* ... */ }

        for element in [10, 20,
        30].iter() { /* ... */ }
        for number in 1..4 { /* ... */ } // Range 1 to 3 (exclusive)
        for number in 1..=4 { /* ... */ } // Range 1 to 4 (inclusive)

5. Ownership & References

// Move semantics
        let s1 = String::from("hello");
        let s2 = s1; //
          s1 is moved to s2, s1 is now invalid

        // Clone
        let s3 = s2.clone(); // Deep copy, s2 is still
          valid

        // References (Borrowing)
        let len = calculate_len(&s3); // Immutable
          reference
        fn calculate_len(s: &String) -> usize { s.len() }

        // Mutable References
        let mut s
        = String::from("hello");
        change(&mut s);
        fn change(some_string: &mut String) { some_string.push_str(", world"); }

        // Slices (Reference to a contiguous sequence)
        let s = String::from("hello world");
        let hello: &str = &s[0..5];
        let world: &str = &s[6..11];
        let slice: &[i32] = &[1, 2,
        3];

6. Structs

// Definition
        struct User {
        username: String,
        email: String,
        sign_in_count: u64,
        active: bool,
        }

        // Instantiation
        let user1 = User {
        email: String::from("someone@example.com"),
        username: String::from("someusername123"),
        active: true,
        sign_in_count: 1,
        };

        // Update syntax
        let user2 = User {
        email: String::from("another@example.com"),
        ..user1 // Copy remaining fields from user1
        };

        // Tuple Structs
        struct Color(i32, i32, i32);
        let black = Color(0, 0, 0);

        struct AlwaysEqual;

7. Enums & Pattern Matching

// Definition
        enum Message {
        Quit,
        Move { x: i32, y: i32 },
        Write(String),
        ChangeColor(i32, i32, i32),
        }

        // Match expression
        let msg = Message::Move { x: 10, y: 20 };
        match msg {
        Message::Quit => println!("Quit"),
        Message::Move { x, y } => println!("Move to {},
          {}", x, y),
        Message::Write(text) => println!("Text: {}", text),
        _ => (), // Catch-all
        }

        // if let (concise control flow)
        if let Message::Write(text) = msg {
        println!("{}", text);
        }

8. Methods & Associated Functions

struct Rectangle { width: u32, height: u32 }

        impl Rectangle {
        // Method (takes &self)
        fn area(&self) -> u32 {
        self.width * self.height
        }

        // Associated function (no self, like static method)
        fn new(width: u32, height: u32) -> Rectangle {
        Rectangle { width, height }
        }
        }

        let rect = Rectangle::new(30, 50);
        rect.area();

9. Error Handling

// Result enum: Ok(T) or Err(E)
        enum Result<T, E> { Ok(T), Err(E)
        }

        // Option enum: Some(T) or None
        enum Option<T> { Some(T), None }

        // Propagation operator (?)
        fn read_username() -> Result<String, io::Error> {
        let mut f
        = File::open("hello.txt")?;
        // Return Err immediately if failed
        let mut s
        = String::new();
        f.read_to_string(&mut
        s)?;
        Ok(s)
        }

        // Unwrapping (Panic on Err/None)
        let f = File::open("hello.txt").unwrap();
        let f = File::open("hello.txt").expect("Failed to open
          file");

11. Lifetimes

// Explicit lifetime annotation
        // 'a relates the lifetime of x, y, and the return value
        fn longest<'a>(x: &'a str, y: &'a str)
        -> &'a str {
        if x.len() > y.len() { x } else { y }
        }

        // Struct with references
        struct ImportantExcerpt<'a> {
        part: &'a str,
        }

        // Static lifetime (lives for entire program duration)
        let s: &'static str = "I have a
          static lifetime.";

12. Closures & Iterators

// Closure syntax
        let add_one = |x: i32| -> i32 {
        x + 1 };
        let plain = |x| x + 1; // Type inference

        // Capturing environment
        let x = 4;
        let equal_to_x = |z| z == x;

        // Move semantics in closures
        let equal_to_x = move |z| z == x; // 'x' is moved into closure

        // Iterators
        let v1 = vec![1, 2,
        3];
        let v2: Vec<_> = v1.iter().map(|x| x + 1).collect();

13. Advanced Pattern Matching

let x =
        1;
        match x {
        1 | 2 => println!("One or Two"), // Or pattern
        1..=5 => println!("1 through 5"), // Range pattern
        _ => println!("Something else"),
        }

        // Destructuring Structs
        let p = Point { x: 0, y: 7 };
        match p {
        Point { x, y: 0 } => println!("On x axis at {}", x),
        Point { x: 0, y } => println!("On y axis at {}", y),
        Point { x, y } => println!("On neither axis:
          ({}, {})", x, y),
        }

        // Ignoring values
        let _ = 5;
        let Point { x, .. } = p; // Ignore y

        // Match Guards
        match Some(4) {
        Some(x) if x < 5 => println!("less
          than five: {}", x),
        _ => (),
        }

        // @ Binding (Test and save value)
        match msg {
        Message::Move { x: val @ 3..=7, y } =>
        println!("x in range: {}", val),
        _ => (),
        }

15. Smart Pointers

// Box<T> (Heap allocation)
        let b = Box::new(5);

        // Rc<T> (Reference Counting - Single Thread)
        use std::rc::Rc;
        let a = Rc::new(5);
        let b = Rc::clone(&a);

        // Arc<T> (Atomic Reference Counting - Thread Safe)
        use std::sync::Arc;
        let a = Arc::new(5);

        // RefCell<T> (Interior Mutability)
        use std::cell::RefCell;
        let x = RefCell::new(5);
        *x.borrow_mut() += 1;
