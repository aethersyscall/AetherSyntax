4. Control Flow

Go has only one looping construct: for.

// If-Else
        if x > 10 {
        fmt.Println("Big")
        } else {
        fmt.Println("Small")
        }

        // If with initialization (variable scope limited to block)
        if val := calculate(); val > 10 {
        fmt.Println(val)
        }

        // Standard For Loop
        for i := 0; i < 5; i++ { /* ... */ }

        // While Loop equivalent
        for x < 100 { /* ... */ }

        // Infinite Loop
        for { break }

        // Range Loop (Slice, Map, Channel)
        nums := []int{1, 2, 3}
        for index, value := range nums {
        fmt.Println(index, value)
        }

        // Switch (No 'break' needed, implied)
        switch os {
        case "darwin":
        fmt.Println("Mac")
        case "linux":
        fmt.Println("Linux")
        default:
        fmt.Println("Other")
        }

5. Arrays & Slices

Arrays are fixed length. Slices are dynamic wrappers around arrays (used 99% of the time).

// Arrays
        var arr [5]int
        // [0 0 0 0 0]
        arr2 := [3]int{1, 2, 3}

        // Slices
        var s []int // nil
          slice
        s = make([]int, 5) // Allocate slice of size 5
        s = make([]int, 5, 10) // Size 5,
          Capacity 10

        // Literal
        nums := []int{1, 2, 3}

        // Operations
        nums = append(nums, 4) // Add element
        val := nums[1] // Access

        // Slicing syntax [low:high] (high is exclusive)
        subset := nums[1:3] // Index 1 up to (not including) 3

        // Operations
        scores["Alice"] = 15 // Update

        // Check existence (Comma ok idiom)

8. Pointers

Go has pointers but no pointer arithmetic.

x := 10
        p := &x // p is a pointer to x
        *p = 20 // dereference and assign
        fmt.Println(x) // 20

        // Pass by value vs Pointer
        func modify(val int) { val = 0 } // No change outside
        func modifyPtr(val *int) { *val = 0 } // Changes original

9. Structs & Methods

Go does not have classes. It uses Structs.

type Vertex struct {
        X, Y int
        }

        v := Vertex{1, 2}
        v.X = 4

        // Methods (Value Receiver)
        // Can read fields, but cannot modify original struct
        func (v
            Vertex) Abs() float64 {
        return math.Sqrt(float64(v.X*v.X + v.Y*v.Y))
        }

        // Methods (Pointer Receiver)
        // Can modify original struct (Preferred for large structs)
        func (v
            *Vertex) Scale(f int) {
        v.X = v.X * f
        v.Y = v.Y * f
        }

10. Interfaces

Interfaces are implemented implicitly. If a struct has the methods, it implements the interface. 

type Abser interface {
        Abs() float64
        }

        // Since *Vertex has Abs(), it implements Abser
        var a Abser = &Vertex{3, 4}

        // Empty Interface (Any type)
        func describe(i interface{}) {
        fmt.Printf("(%v, %T)\n", i, i)
        }

        // Type Assertion
        s, ok := i.(string) // Check if i holds a
          string

11. Error Handling

Go treats errors as values, not exceptions.

import "errors"

        func divide(a, b float64) (float64, error) {
        if b == 0 {
        return 0, errors.New("cannot divide by zero")
        }
        return a / b, nil
        }

        // Usage
        result, err := divide(10, 0)
        if err != nil {
        fmt.Println("Error:", err)
        return
        }
        fmt.Println("Result:", result)

12. Concurrency (Goroutines & Channels)

The "killer feature" of Go.

// Goroutine (Lightweight thread)
        go doSomething()

        // Channels (Typed conduit for data)
        ch := make(chan int)

        // Send/Receive
        go func() { ch <- 42 }() // Send to channel
        val := <-ch // Receive from channel

        // Buffered Channels (Don't block until full)
        ch := make(chan int, 2)

        // Closing
        close(ch)

        // Select (Wait on multiple channel operations)
        select {
        case msg1 := <-c1:
        fmt.Println("Received", msg1)
        case c2 <- "hello":
        fmt.Println("Sent hello")
        default:
        fmt.Println("Non-blocking")
        }

13. Defer, Panic, Recover

// Defer: Schedules function call to run after
          surrounding function returns
        // Used for cleanup (closing files, unlocking mutexes)
        func process() {
        f := openFile()
        defer f.Close() // Will run at end of
          process()
        // ... use f
        }

        // Panic: Aborts execution (Runtime Exception)
        if criticalError {
        panic("Something went wrong")
        }

        // Recover: Regain control of a panicking goroutine
        func safe() {
        defer func() {
        if r := recover(); r != nil {
        fmt.Println("Recovered from:", r)
        }
        }()
        panic("Crash")
        }

14. JSON Handling

Standard library support via encoding/json.

type User struct {
        Name string `json:"username"` // Struct Tag
        Age int `json:"age"`
        Email string `json:"-"` // Ignore field
        }

        // Marshal (Struct -> JSON)
        u := User{Name: "Bob", Age: 30}
        jsonData, _ := json.Marshal(u)

        // Unmarshal (JSON -> Struct)
        var u2 User
        json.Unmarshal(jsonData, &u2)
